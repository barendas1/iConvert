Item Category Query:

  <WebcreteXMLMsgsRq>
    <ItemCategoryQueryRq>
    </ItemCategoryQueryRq>
  </WebcreteXMLMsgsRq>
</WebcreteXML>

Material Query:

<?webcretexml version="1.0"?>
<WebcreteXML>
  <WebcreteXMLMsgsRq>
    <ItemQueryRq>
      <IncludeRetElement>Location</IncludeRetElement>
      <IncludeRetElement>Cost</IncludeRetElement>
      <IncludeRetElement>Pricing</IncludeRetElement>
      <IncludeRetElement>Batching</IncludeRetElement>
